pragma solidity ^0.8.20;
 
import "@openzeppelin/contracts/metatx/ERC2771Forwarder.sol";
import "@openzeppelin/contracts/access/Ownable2Step.sol";
 
/// @title KGenForwarder
/// @notice A meta-transaction forwarder that only allows trusted relayers to execute transactions
/// @dev Extends OpenZeppelin's ERC2771Forwarder with relayer access control
contract KGenForwarder is ERC2771Forwarder, Ownable2Step {
    mapping(address => bool) public trustedRelayers;
 
    event TrustedRelayerAdded(address indexed relayer);
    event TrustedRelayerRemoved(address indexed relayer);
 
    /// @notice Constructs the forwarder with a domain name and sets the initial owner
    /// @param domain_name The domain name used for EIP-712 signing
    constructor(
        string memory domain_name
    ) ERC2771Forwarder(domain_name) Ownable(msg.sender) {}
 
    /// @notice Adds a new trusted relayer
    /// @param relayer The address to be added as a trusted relayer
    /// @dev Only callable by the contract owner
    function addTrustedRelayer(address relayer) external onlyOwner {
        trustedRelayers[relayer] = true;
        emit TrustedRelayerAdded(relayer);
    }
 
    /// @notice Removes a trusted relayer
    /// @param relayer The address to be removed from trusted relayers
    /// @dev Only callable by the contract owner
    function removeTrustedRelayer(address relayer) external onlyOwner {
        trustedRelayers[relayer] = false;
        emit TrustedRelayerRemoved(relayer);
    }
 
    /// @notice Executes a meta-transaction if sent by a trusted relayer
    /// @param request The forward request data containing the meta-transaction details
    /// @dev Overrides the execute function from ERC2771Forwarder to add relayer verification
    function execute(
        ForwardRequestData calldata request
    ) public payable override {
        if (!trustedRelayers[msg.sender]) {
            revert("Not a trusted relayer");
        }
        super.execute(request);
    }
 
    function recoverForwardRequestSigner(
        ForwardRequestData calldata request
    ) public view returns (bool, address) {
        return _recoverForwardRequestSigner(request);
    }
}
